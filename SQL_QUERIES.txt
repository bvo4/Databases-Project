CREATE TABLE users (
  uid INT NOT NULL auto_increment primary key,
  username VARCHAR(20) unique NOT NULL,
  password VARCHAR(30) NOT NULL,
  profile VARCHAR(99),
  status VARCHAR(15) NOT NULL,
  city VARCHAR(25) NOT NULL,
  state VARCHAR(2) NOT NULL,
  country VARCHAR(2) NOT NULL
  );
CREATE TABLE answers (
  aid INT NOT NULL primary key,
  body VARCHAR(99) NOT NULL
  );
CREATE TABLE topic (
  tid INT NOT NULL primary key,
  tname VARCHAR(50),
  hierarchy INT NOT NULL);
CREATE TABLE subtopic (
  stid INT NOT NULL primary key,
  tid INT references topic(tid),
  sname VARCHAR(50)
  );
CREATE TABLE  questions (
  qid INT NOT NULL primary key,
  stid INT references subtopic(stid),
  title VARCHAR(60) NOT NULL,
  body VARCHAR(90) 
  );
CREATE TABLE likes (
  aid int references answers(aid),
  uid int references users(uid),
  points int
  );
CREATE TABLE post_question (
  qid INT references questions(qid),
  uid INT references users(uid),
  resolved BOOLEAN,
  timeposted DATETIME NOT NULL
  );
CREATE TABLE post_answers (
  uid INT references users(uid),
  qid INT references questions(qid),
  aid INT references answers(aid),
  grade int not null,
  weight int not null,
  best boolean not null,
  timeposted DATETIME NOT NULL
  );
